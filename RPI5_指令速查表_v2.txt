【第五步】HTTPS 憑證設定（全部在 RPI5 上執行，只需一次）
----------------------------------------------------------------

# ─── Step 1: 安裝 certbot ───
sudo apt update && sudo apt install -y certbot


# ─── Step 2: 取得 DuckDNS token ───
# 瀏覽器登入 https://www.duckdns.org
# 複製頁面上方顯示的 token（格式：xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx）


# ─── Step 3: 執行 certbot 申請憑證 ───
# email 請換成你自己的
sudo certbot certonly \
  --manual \
  --preferred-challenges dns \
  --agree-tos \
  --email your@email.com \
  -d rpi5demo.duckdns.org

# 執行後 certbot 會暫停，顯示一串驗證碼，例如：
#   with the following value:
#   ABC123xyzDEF456uvw   ← 複製這串
#   Press Enter to Continue   ← 先不要按！


# ─── Step 4: 更新 DuckDNS TXT 紀錄（關鍵！）───
# DuckDNS 網頁介面「沒有」TXT 編輯欄位，必須用 URL API 更新
# 在瀏覽器打開以下網址（替換 token 和 txt 兩個參數）：

https://www.duckdns.org/update?domains=rpi5demo&token=你的token&txt=certbot給的驗證碼

# 瀏覽器顯示 OK 代表更新成功
# 若顯示 KO 請檢查 token 和 domains 是否正確


# ─── Step 5: 等 DNS 傳播，回 RPI5 按 Enter ───
# 建議等 10-20 秒，可以另開終端確認 TXT 是否已生效：
nslookup -type=TXT _acme-challenge.rpi5demo.duckdns.org 8.8.8.8

# 看到驗證碼出現後，回到 certbot 終端按 Enter


# ─── Step 6: 確認憑證產生成功 ───
sudo ls -lh /etc/letsencrypt/live/rpi5demo.duckdns.org/

# 檢查有效期
sudo openssl x509 -enddate -noout \
  -in /etc/letsencrypt/live/rpi5demo.duckdns.org/fullchain.pem


# ─── Step 7: 設定 HTTPS 服務 ───
# 先修改 setup_https.sh 裡的網域名稱
nano /home/p400/rpi5-demo/setup_https.sh
# 修改：DEMO_DOMAIN="rpi5demo.duckdns.org"

sudo bash /home/p400/rpi5-demo/setup_https.sh


----------------------------------------------------------------
【憑證更新】Let's Encrypt 憑證 90 天後需更新
----------------------------------------------------------------

# 檢查剩餘天數
sudo openssl x509 -enddate -noout \
  -in /etc/letsencrypt/live/rpi5demo.duckdns.org/fullchain.pem

# 重新申請（與初次申請指令相同）
sudo certbot certonly \
  --manual \
  --preferred-challenges dns \
  --agree-tos \
  --email your@email.com \
  -d rpi5demo.duckdns.org
# certbot 會偵測到現有憑證，選 "2: Renew & replace the certificate"
# 後續流程與初次申請相同：開瀏覽器更新 TXT、等 10 秒、按 Enter

# 更新後重啟 nginx
sudo systemctl restart nginx

# 💡 建議：加行事曆提醒每 60 天檢查，提早 30 天更新避免過期

【問題排除】
----------------------------------------------------------------

# Port 8000 已被佔用 → 停掉背景服務
sudo systemctl stop rpi5-demo
sudo fuser -k 8000/tcp

# WiFi 熱點沒啟動
sudo nmcli connection up rpi5-hotspot

# nginx 設定有問題
sudo nginx -t
sudo systemctl status nginx

# 憑證快到期 → 詳見上方「憑證更新」章節

# 查看 RPI5 IP
hostname -I

# 清除 Journal Log（SD 卡空間不足時）
sudo journalctl --vacuum-size=50M
